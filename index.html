async function decodificar() { const inputFull = document.getElementById('input-code').value.trim().toUpperCase(), errorMsg = document.getElementById('error-msg'), resBox = document.getElementById('result'); errorMsg.innerText = ""; resBox.style.display = "none"; if (inputFull.length < 2) { errorMsg.innerText = "Código inválido."; return; } // Ocultar todas las filas dinámicas al iniciar la búsqueda ['row-conf', 'row-sal', 'row-ang', 'row-jun', 'row-ent', 'row-maq', 'row-flu', 'res-fam'].forEach(id => { let el = document.getElementById(id); if(el) el.style.display = "none"; }); document.getElementById('res-fam').parentElement.style.display = "flex"; // Mostrar fila contenedora de Familia // --- ESCENARIO 1: ES UN CÓDIGO EXCLUSIVO DE MÁQUINA (Ej: I1-10032) --- if (inputFull.startsWith("I") || inputFull.startsWith("O")) { document.getElementById('res-fam').innerText = "CONEXIÓN DIRECTA DE MÁQUINA"; document.getElementById('res-fam').style.display = "block"; let resultadoBD = await consultarIdBD(inputFull); if (resultadoBD.status === "error") { errorMsg.innerText = resultadoBD.message; return; } if (resultadoBD.isMachine) { document.getElementById('res-maq').innerText = resultadoBD.maquina; document.getElementById('res-flu').innerText = resultadoBD.flujo.toUpperCase(); document.getElementById('row-maq').style.display = "flex"; document.getElementById('row-flu').style.display = "flex"; resBox.style.display = "block"; } return; // Finaliza el proceso para el caso de máquinas } // --- ESCENARIO 2: CÓDIGO ESTÁNDAR (Ej: FLAA-10 o CTBA-1-2) --- const parts = inputFull.split('-'); const head = parts[0]; if (head.length < 3 && head.substring(0,2) !== "TU") { errorMsg.innerText = "Formato de código estándar inválido."; return; } const fam = head.substring(0, 2); if (!famMap[fam]) { errorMsg.innerText = "Familia no reconocida."; return; } if (parts.length < 2) { errorMsg.innerText = "Faltan las variables de dimensión en el código."; return; } document.getElementById('res-fam').innerText = famMap[fam]; document.getElementById('res-fam').style.display = "block"; try { let idx = 2, conf = ""; if (fam !== "TU") { conf = head.charAt(idx++); document.getElementById('res-conf').innerText = (dataRef[fam] && dataRef[fam].conf && dataRef[fam].conf[conf]) ? dataRef[fam].conf[conf] : "Desconocido"; document.getElementById('row-conf').style.display = "flex"; } const meVal = matEspRev[head.charAt(idx++)] || "Desconocido"; document.getElementById('res-mat').innerText = meVal.split(' (')[0]; document.getElementById('res-esp').innerText = meVal.includes('(') ? meVal.split('(')[1].replace(')', '') : "Desconocido"; if (fam !== "FL") { document.getElementById('res-jun').innerText = junMapDec[head.charAt(idx++)] || "Desconocido"; document.getElementById('row-jun').style.display = "flex"; } if (fam === "CE" || fam === "CF") { let angStr = head.substring(idx); document.getElementById('res-ang').innerText = angStr + "°"; document.getElementById('row-ang').style.display = "flex"; } let resultadoBD = await consultarIdBD(parts[1]); if (resultadoBD.status === "error") { errorMsg.innerText = "Dimensión Entrada: " + resultadoBD.message; return; } document.getElementById('res-ent').innerText = resultadoBD.medida + " mm"; document.getElementById('row-ent').style.display = "flex"; if (fam === "CT" && parts.length >= 3) { let resultadoSalida = await consultarIdBD(parts[2]); if (resultadoSalida.status === "error") { errorMsg.innerText = "Dimensión Salida: " + resultadoSalida.message; return; } document.getElementById('res-sal').innerText = resultadoSalida.medida + " mm"; document.getElementById('row-sal').style.display = "flex"; } resBox.style.display = "block"; } catch (e) { errorMsg.innerText = "Error crítico de procesamiento."; } }
